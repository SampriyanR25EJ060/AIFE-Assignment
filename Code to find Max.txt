#include <stdio.h>

// Function to find the maximum value in an array
int findMaximum(int arr[], int size) {
    // Start by assuming the first element is the largest
    int max = arr[0]; 
    
    // Loop through the rest of the array elements
    for (int i = 1; i < size; i++) {
        // If a larger element is found, update the max variable
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    
    return max;
}

int main() {
    // Example array
    int numbers[] = {3, 7, 2, 9, 5, 14, 8};
    
    // Calculate the number of elements in the array
    int size = sizeof(numbers) / sizeof(numbers[0]);
    
    // Call the function and store the result
    int max_value = findMaximum(numbers, size);
    
    // Print the result
    printf("The maximum value in the array is: %d\n", max_value);
    
    return 0;
}
